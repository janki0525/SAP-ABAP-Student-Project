REPORT ZSTUDENT_REPORT.
 TABLES: ZSTUDENT_MASTER.

  DATA: IT_STUDENTS TYPE TABLE OF
  ZSTUDENT_MASTER,
  IS_STUDENT TYPE ZSTUDENT_MASTER.

  SELECT * FROM ZSTUDENT_MASTER INTO TABLE
    IT_STUDENTS.

    IF SY-SUBRC <> 0.
      WRITE: 'NO DATA FOUND'.
      ELSE.

        WRITE: / 'STUDENT ID', 15 'NAME', 35 'AGE', 50 'COURSE ID'.
        ULINE.

        LOOP AT IT_STUDENTS INTO IS_STUDENT.

          WRITE: / IS_STUDENT-STUDENT_ID,
                   15 IS_STUDENT-NAME,
                   35 IS_STUDENT-AGE,
                   50 IS_STUDENT-COURSE_ID.

          ENDLOOP.

          ENDIF.
